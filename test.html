$(function () { $("#userinfo-mobile").popover({ template: '
', html: true, trigger: "hover", placement: "bottom", delay: {hide: 100} }).on('shown.bs.popover', function (event) { var that = this; $(this).parent().find('div.popover').on('mouseenter', function () { $(that).attr('in', true); }).on('mouseleave', function () { $(that).removeAttr('in'); $(that).popover('hide'); }); }).on('hide.bs.popover', function (event) { if ($(this).attr('in')) { event.preventDefault(); } }); })
